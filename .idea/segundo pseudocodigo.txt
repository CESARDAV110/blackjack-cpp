			// Inicializar puntajes

			carta_computadora <- Aleatorio(1, 10)

			// Turno del jugador
			Mientras puntaje_jugador < 21 Hacer
				Escribir "¿Quieres pedir otra carta? (S/N)"
				Leer respuesta
				Si respuesta = 'S' o respuesta = 's' Entonces
					carta_jugador <- Aleatorio(1, 10)
					puntaje_jugador <- puntaje_jugador + carta_jugador

					// Verificación de puntaje del jugador
					Si puntaje_jugador > 21 Entonces

					FinSi
					
					Si puntaje_jugador = 21 Entonces
						Escribir "¡El jugador ha ganado con 21!"
					FinSi
				SiNo
				FinSi
			FinMientras
			
			// Turno de la computadora (si el jugador no ha perdido)
			Si puntaje_jugador <= 21 Entonces

				Mientras puntaje_computadora < 17 Hacer
					carta_computadora <- Aleatorio(1, 10)

				FinMientras
